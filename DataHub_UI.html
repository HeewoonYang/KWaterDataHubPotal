-- 이상치 제거 + 단위 통일 변환

WITH source AS (
  SELECT
    obsv_stn_id,
    obsv_dt,
    water_lvl,
    meas_mthd_cd
  FROM {{ source('rwis', 'tb_water') }}
)

SELECT
  obsv_stn_id,
  obsv_dt,
  CASE
    WHEN water_lvl < 0 THEN NULL
    WHEN water_lvl > 200 THEN NULL
    ELSE ROUND(water_lvl, 2)
  END AS water_level_m,
  COALESCE(meas_mthd_cd, 'UNKNOWN')
FROM source

DROP TABLE IF EXISTS DM_WATER_LEVEL CASCADE;

CREATE TABLE DM_WATER_LEVEL (
    obs_id         BIGINT          NOT NULL,
    dam_id         VARCHAR(10)     NOT NULL,
    water_level    DECIMAL(8,2),
    measured_at    TIMESTAMP,
    measure_method VARCHAR(10),
    CONSTRAINT PK_DM_WATER_LEVEL PRIMARY KEY (obs_id),
    CONSTRAINT FK_WL_DAM FOREIGN KEY (dam_id) REFERENCES DM_DAM_INFO(dam_id)
);

COMMENT ON TABLE DM_WATER_LEVEL IS '실시간 수위 관측 데이터';

-- -----------------------------------------

DROP TABLE IF EXISTS DM_FLOW_RATE CASCADE;

CREATE TABLE DM_FLOW_RATE (
    flow_id      BIGINT          NOT NULL,
    dam_id       VARCHAR(10)     NOT NULL,
    flow_rate    DECIMAL(10,3),
    measured_at  TIMESTAMP,
    CONSTRAINT PK_DM_FLOW_RATE PRIMARY KEY (flow_id),
    CONSTRAINT FK_FR_DAM FOREIGN KEY (dam_id) REFERENCES DM_DAM_INFO(dam_id)
);

COMMENT ON TABLE DM_FLOW_RATE IS '유입량/방류량 측정 기록';

-- -----------------------------------------

DROP TABLE IF EXISTS DM_DISCHARGE CASCADE;

CREATE TABLE DM_DISCHARGE (
    discharge_id BIGINT          NOT NULL,
    dam_id       VARCHAR(10)     NOT NULL,
    volume       DECIMAL(12,2),
    start_dt     TIMESTAMP,
    end_dt       TIMESTAMP,
    CONSTRAINT PK_DM_DISCHARGE PRIMARY KEY (discharge_id),
    CONSTRAINT FK_DC_DAM FOREIGN KEY (dam_id) REFERENCES DM_DAM_INFO(dam_id)
);

COMMENT ON TABLE DM_DISCHARGE IS '방류 시행 및 계획 이력';

DROP TABLE IF EXISTS DM_WATER_QUALITY CASCADE;

    quality_id   BIGINT          NOT NULL,

    ph           DECIMAL(4,2),
    turbidity    DECIMAL(8,2),
    do_value     DECIMAL(6,2),
    CONSTRAINT PK_DM_WATER_QUALITY PRIMARY KEY (quality_id),
    CONSTRAINT FK_WQ_DAM FOREIGN KEY (dam_id) REFERENCES DM_DAM_INFO(dam_id)
);

COMMENT ON TABLE DM_WATER_QUALITY IS '수질 항목별 분석 결과';
