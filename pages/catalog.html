// 수자원시설별 연결 시스템 및 데이터셋 전체 조회
MATCH (f:WaterFacility)
  -[:HAS_SYSTEM]->(s:System)
  -[:PRODUCES]->(d:DataSet)
WHERE f.facilityType IN ['댐', '보', '정수장']
RETURN f.facilityName AS facility,
    f.facilityType AS type,
    s.name AS system,
    d.name AS dataset,
    d.recordCount AS records,
    d.qualityScore AS quality
ORDER BY d.qualityScore DESC
LIMIT 20
